      | MATH '-' TERMO
      | TERMO

       | TERMO '/' FACTOR
       | FACTOR

        | '(' MATH ')'
        | VALOR

          | VAZIO

      | VALOR1 '>' VALOR1 COND1
      | VALOR1 '=' VALOR1 COND1
      | VALOR1 '<=' VALOR1 COND1
      | VALOR1 '>=' VALOR1 COND1

 COND1 : '|''|' COND
       | '&''&' COND
